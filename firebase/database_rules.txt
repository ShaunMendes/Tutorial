rules_version = '2';
service cloud.firestore {                          //apply only to cloud firebase
  match /databases/{database}/documents {          //apply to all database. Specify database by changing {database}
    match /{document=**} {                         //apply to all documents
      allow read, write;
    }
  }
}

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /projects/{project} {  //for each document in projects
      allow read, write: if request.auth.uid != null 
    }
    match /users/{userId}{
    allow create
    allow read: if request.auth.uid != null
    allow write: if request.auth.uid == userId
    }
  }
}
